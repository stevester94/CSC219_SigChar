########################################################################
# Setup library
########################################################################
include(GrPlatform) #define LIB_SUFFIX

include_directories(${Boost_INCLUDE_DIR})
link_directories(${Boost_LIBRARY_DIRS})

list(APPEND basic_sources
)

set(basic_sources "${basic_sources}" PARENT_SCOPE)
if(NOT basic_sources)
	MESSAGE(STATUS "No C++ sources... skipping lib/")
	return()
endif(NOT basic_sources)

add_library(gnuradio-basic SHARED ${basic_sources})
target_link_libraries(gnuradio-basic ${Boost_LIBRARIES} ${GNURADIO_ALL_LIBRARIES})
set_target_properties(gnuradio-basic PROPERTIES DEFINE_SYMBOL "gnuradio_basic_EXPORTS")

if(APPLE)
    set_target_properties(gnuradio-basic PROPERTIES
        INSTALL_NAME_DIR "${CMAKE_INSTALL_PREFIX}/lib"
    )
endif(APPLE)

########################################################################
# Install built library files
########################################################################
include(GrMiscUtils)
GR_LIBRARY_FOO(gnuradio-basic RUNTIME_COMPONENT "basic_runtime" DEVEL_COMPONENT "basic_devel")

########################################################################
# Build and register unit test
########################################################################
include(GrTest)

include_directories(${CPPUNIT_INCLUDE_DIRS})

list(APPEND test_basic_sources
    ${CMAKE_CURRENT_SOURCE_DIR}/test_basic.cc
    ${CMAKE_CURRENT_SOURCE_DIR}/qa_basic.cc
)

add_executable(test-basic ${test_basic_sources})

target_link_libraries(
  test-basic
  ${GNURADIO_RUNTIME_LIBRARIES}
  ${Boost_LIBRARIES}
  ${CPPUNIT_LIBRARIES}
  gnuradio-basic
)

GR_ADD_TEST(test_basic test-basic)

########################################################################
# Print summary
########################################################################
message(STATUS "Using install prefix: ${CMAKE_INSTALL_PREFIX}")
message(STATUS "Building for version: ${VERSION} / ${LIBVER}")
